# FYS-plan (Fysisk treningsplan) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bygg et komplett FYS-treningssystem i PlayerHQ — planlegger, logging, tester, analyse og coach-tildeling — som matcher strukturen til Teknisk Plan og er fullt synkronisert med testresultater og analysedelen.

**Architecture:** Ny modell-hierarki `FysiskPlan → FysUke → FysOkt → FysOvelseRad`. Coach kan opprette `FysPlanTemplate` og tildele planer til spillere (samme godkjenningsflyt som TrainingPlan). `FysUke` har periodiseringstype (AKKUMULERING/INTENSIVERING/REALISERING/DELOAD). `FysOkt` har økt-kompletering, hel-økt-RPE og notat. `FysOvelseRad` har tempo-notasjon og bevegelsesmønster. Kroppslogg-modell (`KroppsMaling`) for relativstyrke-beregning. Analyse-integrasjon: ACWR-chart, golf-korrelasjon (styrke vs CHS), muskelgruppe-balanse, PR-deteksjon, 1RM-sync fra TestResult.

**Tech Stack:** Next.js 16 App Router, Prisma 7, Supabase (SQL via MCP), Tailwind CSS v4, lucide-react, TypeScript strict

```prisma
// ============================================================================
// FysiskPlan — Fysisk treningsplan (styrke/kondisjon)
// Hierarki: FysiskPlan → FysUke → FysOkt → FysOvelseRad
// ============================================================================

model FysiskPlan {
  id            String         @id @default(cuid())
  userId        String         // spiller
  opprettetAvId String         // coach eller spiller
  navn          String         // "FYS-plan vår 2026"
  status        TechPlanStatus @default(DRAFT)
  startDato     DateTime
  sluttDato     DateTime?
  notater       String?        // overordnet beskrivelse av planen

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  user        User     @relation("FysiskPlanUser", fields: [userId], references: [id], onDelete: Cascade)
  opprettetAv User     @relation("FysiskPlanCreator", fields: [opprettetAvId], references: [id])
  uker        FysUke[]

  @@index([userId, status])
  @@map("fysiske_planer")
}

model FysUke {
  id        String @id @default(cuid())
  planId    String
  ukeNr     Int    // 1, 2, 3 …
  label     String // "Uke 1 · Akkumulering"
  sortOrder Int

  plan  FysiskPlan @relation(fields: [planId], references: [id], onDelete: Cascade)
  okter FysOkt[]

  @@unique([planId, ukeNr])
  @@index([planId, sortOrder])
  @@map("fys_uker")
}

model FysOkt {
  id               String   @id @default(cuid())
  ukeId            String
  navn             String   // "Økt A · Underkropp"
  dag              Ukedag?  // MAN, TIR … SON
  sortOrder        Int
  estimertMinutter Int?     // estimert varighet i minutter

  uke   FysUke        @relation(fields: [ukeId], references: [id], onDelete: Cascade)
  rader FysOvelseRad[]

  @@index([ukeId, sortOrder])
  @@map("fys_okter")
}

model FysOvelseRad {
  id         String  @id @default(cuid())
  oktId      String
  exerciseId String? // valgfri ref til ExerciseDefinition
  navn       String  // visningsnavn (kopiert fra exercise eller fri tekst)
  sortOrder  Int

  // Planlagte treningsparametre
  sett          Int     @default(3)   // antall sett
  repsMin       Int?                  // min reps (f.eks. 8 i «8-12»)
  repsMax       Int?                  // max reps (f.eks. 12 i «8-12»)
  hvile         Int?                  // hvile mellom sett i sekunder
  belastningPst Int?                  // belastning i % av 1RM (f.eks. 75)
  rir           Int?                  // reps i reserve (f.eks. 2)
  muskelgruppe  String?               // primær muskelgruppe (fri tekst)

  // Logging — faktisk utført
  loggSett         Int?
  loggRepsPerSett  String?  // «8,10,10» — kommaseparert per sett
  loggBelastningKg Float?   // faktisk vekt i kg
  loggRir          Int?     // faktisk RIR etter siste sett
  notat            String?  // spiller-kommentar

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  okt      FysOkt              @relation(fields: [oktId], references: [id], onDelete: Cascade)
  exercise ExerciseDefinition? @relation(fields: [exerciseId], references: [id], onDelete: SetNull)

  @@index([oktId, sortOrder])
  @@map("fys_ovelserader")
}
```

```sql
DO $$ BEGIN
  CREATE TYPE "Ukedag" AS ENUM ('MAN','TIR','ONS','TOR','FRE','LOR','SON');
EXCEPTION WHEN duplicate_object THEN NULL;
END $$;

-- fysiske_planer
CREATE TABLE IF NOT EXISTS fysiske_planer (
  id              TEXT PRIMARY KEY,
  "userId"        TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  "opprettetAvId" TEXT NOT NULL REFERENCES users(id),
  navn            TEXT NOT NULL,
  status          "TechPlanStatus" NOT NULL DEFAULT 'DRAFT',
  "startDato"     TIMESTAMPTZ NOT NULL,
  "sluttDato"     TIMESTAMPTZ,
  notater         TEXT,
  "createdAt"     TIMESTAMPTZ NOT NULL DEFAULT now(),
  "updatedAt"     TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_fysiske_planer_user_status ON fysiske_planer("userId", status);

-- fys_uker
CREATE TABLE IF NOT EXISTS fys_uker (
  id         TEXT PRIMARY KEY,
  "planId"   TEXT NOT NULL REFERENCES fysiske_planer(id) ON DELETE CASCADE,
  "ukeNr"    INTEGER NOT NULL,
  label      TEXT NOT NULL,
  "sortOrder" INTEGER NOT NULL,
  UNIQUE("planId", "ukeNr")
);
CREATE INDEX IF NOT EXISTS idx_fys_uker_plan_sort ON fys_uker("planId", "sortOrder");

-- fys_okter
CREATE TABLE IF NOT EXISTS fys_okter (
  id                  TEXT PRIMARY KEY,
  "ukeId"             TEXT NOT NULL REFERENCES fys_uker(id) ON DELETE CASCADE,
  navn                TEXT NOT NULL,
  dag                 "Ukedag",
  "sortOrder"         INTEGER NOT NULL,
  "estimertMinutter"  INTEGER
);
CREATE INDEX IF NOT EXISTS idx_fys_okter_uke_sort ON fys_okter("ukeId", "sortOrder");

-- fys_ovelserader
CREATE TABLE IF NOT EXISTS fys_ovelserader (
  id                TEXT PRIMARY KEY,
  "oktId"           TEXT NOT NULL REFERENCES fys_okter(id) ON DELETE CASCADE,
  "exerciseId"      TEXT REFERENCES exercise_definitions(id) ON DELETE SET NULL,
  navn              TEXT NOT NULL,
  "sortOrder"       INTEGER NOT NULL,
  sett              INTEGER NOT NULL DEFAULT 3,
  "repsMin"         INTEGER,
  "repsMax"         INTEGER,
  hvile             INTEGER,
  "belastningPst"   INTEGER,
  rir               INTEGER,
  muskelgruppe      TEXT,
  "loggSett"        INTEGER,
  "loggRepsPerSett" TEXT,
  "loggBelastningKg" DOUBLE PRECISION,
  "loggRir"         INTEGER,
  notat             TEXT,
  "createdAt"       TIMESTAMPTZ NOT NULL DEFAULT now(),
  "updatedAt"       TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_fys_ovelserader_okt_sort ON fys_ovelserader("oktId", "sortOrder");
```

```typescript
import { Dumbbell } from "lucide-react";

export interface OvelseRadData {
  id: string;
  navn: string;
  muskelgruppe: string | null;
  sett: number;
  repsMin: number | null;
  repsMax: number | null;
  hvile: number | null;          // sekunder
  belastningPst: number | null;  // % av 1RM
  rir: number | null;
  // Logging
  loggSett: number | null;
  loggRepsPerSett: string | null;
  loggBelastningKg: number | null;
  loggRir: number | null;
  notat: string | null;
}

function formatReps(min: number | null, max: number | null): string {
  if (!min && !max) return "—";
  if (min && max && min !== max) return `${min}–${max}`;
  return String(min ?? max ?? "—");
}

function formatHvile(sek: number | null): string {
  if (!sek) return "—";
  if (sek < 60) return `${sek}s`;
  const min = Math.floor(sek / 60);
  const rest = sek % 60;
  return rest === 0 ? `${min} min` : `${min}m${rest}s`;
}

export function OvelseTabell({ rader }: { rader: OvelseRadData[] }) {
  if (rader.length === 0) {
    return (
      <div className="fys-empty-state" style={{ padding: "24px 16px" }}>
        <div className="ic"><Dumbbell size={18} /></div>
        <div>
          <span className="ttl">Ingen øvelser ennå</span>
          <span className="desc">Legg til øvelser i denne økten.</span>
        </div>
      </div>
    );
  }

  return (
    <div style={{ overflowX: "auto" }}>
      <table className="ovelse-tabell">
        <thead>
          <tr>
            <th style={{ width: "32px" }}>#</th>
            <th>Øvelse</th>
            <th>Sett</th>
            <th>Reps</th>
            <th>Hvile</th>
            <th>% 1RM</th>
            <th>RIR</th>
            <th>Logg</th>
          </tr>
        </thead>
        <tbody>
          {rader.map((rad, i) => (
            <tr key={rad.id}>
              <td>
                <span className="ovelse-chip" style={{ minWidth: "28px", justifyContent: "center" }}>
                  {i + 1}
                </span>
              </td>
              <td>
                <div className="ovelse-navn">{rad.navn}</div>
                {rad.muskelgruppe && (
                  <div className="ovelse-muskel">{rad.muskelgruppe}</div>
                )}
              </td>
              <td>
                <span className="ovelse-chip">{rad.sett}</span>
              </td>
              <td>
                <span className="ovelse-chip">{formatReps(rad.repsMin, rad.repsMax)}</span>
              </td>
              <td>
                <span className="ovelse-chip hvile">{formatHvile(rad.hvile)}</span>
              </td>
              <td>
                {rad.belastningPst != null ? (
                  <span className="ovelse-chip belastning">{rad.belastningPst}%</span>
                ) : (
                  <span style={{ color: "hsl(var(--muted-foreground))", fontSize: "11px" }}>—</span>
                )}
              </td>
              <td>
                {rad.rir != null ? (
                  <span className="ovelse-chip rir">RIR {rad.rir}</span>
                ) : (
                  <span style={{ color: "hsl(var(--muted-foreground))", fontSize: "11px" }}>—</span>
                )}
              </td>
              <td>
                {rad.loggSett != null ? (
                  <div style={{ fontSize: "11px" }}>
                    <div style={{ fontWeight: 700 }}>{rad.loggSett} sett</div>
                    {rad.loggRepsPerSett && (
                      <div style={{ color: "hsl(var(--muted-foreground))" }}>{rad.loggRepsPerSett} reps</div>
                    )}
                    {rad.loggBelastningKg != null && (
                      <div style={{ color: "hsl(var(--muted-foreground))" }}>{rad.loggBelastningKg} kg</div>
                    )}
                  </div>
                ) : (
                  <span style={{ color: "hsl(var(--muted-foreground))", fontSize: "11px" }}>Ikke logget</span>
                )}
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

// Volum-kalkulator: sett × reps × kg (estimert fra % 1RM med placeholder 100kg)
export function beregnVolum(rader: OvelseRadData[]): number {
  return rader.reduce((sum, r) => {
    const reps = r.repsMin ?? r.repsMax ?? 0;
    const kg = r.loggBelastningKg ?? (r.belastningPst ? r.belastningPst : 0);
    const sett = r.loggSett ?? r.sett;
    return sum + sett * reps * kg;
  }, 0);
}
```
